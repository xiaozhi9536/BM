⚠
腳位衝突修正
MAX7219 CLK = D6，WS2812 原始範例也是 D6 → 本文所有程式已將 WS2812 改為 D8。
另：AHT10 VIN 只能接 3.3V，接 5V 會損毀晶片。

Arduino C++ · AHT10 + BH1750 
// ── AHT10（溫濕度）+ BH1750（光照）+ MAX7219 + WS2812 ──
// AHT10 VIN 接 3.3V！  WS2812 Data = D8（避免與 MAX CLK=D6 衝突）

#include <Wire.h>
#include <AHT10.h>
#include <BH1750.h>
#include <LedControl.h>
#include <BMduino_WS2812.h>

// MAX7219: Data=D7, CLK=D6, CS=D5, 1 顆
LedControl lc = LedControl(7, 6, 5, 1);
// WS2812: 10 顆，資料腳位 D8
BMduino_WS2812 BW(10, 8);

AHT10   myAHT10(AHT10_ADDRESS_0X38);
BH1750  lightMeter;

float    temp, humidity;
uint16_t lux;

// 顯示溫度 XX.X，佔 pos、pos-1、pos-2（小數點）、pos-3
void dispTemp(int pos, float val) {
  int v = (int)(val * 10 + 0.5);          // ×10 四捨五入
  lc.setChar(0, pos,   ' ', false);       // 最高位空白
  lc.setDigit(0, pos-1, (v/100)%10, false);
  lc.setDigit(0, pos-2, (v/10)%10,  true); // 個位加小數點
  lc.setDigit(0, pos-3, v%10,        false);
}

// 顯示整數（最多 4 位），佔 pos ~ pos-3
void dispInt(int pos, int val) {
  if (val > 9999) val = 9999;
  lc.setDigit(0, pos,   val/1000,       false);
  lc.setDigit(0, pos-1, (val/100)%10, false);
  lc.setDigit(0, pos-2, (val/10)%10,  false);
  lc.setDigit(0, pos-3, val%10,        false);
}

void setup() {
  Serial.begin(9600);
  Wire.begin();

  lc.shutdown(0, false);
  lc.setIntensity(0, 9);
  lc.clearDisplay(0);
  BW.begin();

  while (myAHT10.begin() != true) {
    Serial.println("AHT10 not connected...");
    delay(5000);
  }
  lightMeter.begin();
  Serial.println("Init OK");
}

void loop() {
  temp     = myAHT10.readTemperature();
  humidity = myAHT10.readHumidity();
  lux      = lightMeter.readLightLevel();

  Serial.print("Temp:");  Serial.print(temp);
  Serial.print("C  Hum:"); Serial.print(humidity);
  Serial.print("%  Lux:"); Serial.println(lux);

  // MAX7219：左 4 位 = 溫度，右 4 位 = 光照
  dispTemp(7, temp);   // 位置 7,6,5,4
  dispInt(3,  lux);    // 位置 3,2,1,0

  // WS2812：暗=藍  中=綠  亮=紅
  if (lux < 100) {
    BW.fill(0, 10, 0,   0,   255); // 藍
  } else if (lux < 500) {
    BW.fill(0, 10, 0,   255, 0);   // 綠
  } else {
    BW.fill(0, 10, 255, 0,   0);   // 紅
  }
  BW.show();
  delay(1000);
}

Arduino C++ · AHT10 + HC-SR04 
// ── AHT10（溫濕度）+ HC-SR04（超音波）+ MAX7219 + WS2812 ──
// AHT10 VIN 接 3.3V！  HC-SR04: Trig=D13, Echo=D12

#include <Wire.h>
#include <AHT10.h>
#include <LedControl.h>
#include <BMduino_WS2812.h>

LedControl    lc = LedControl(7, 6, 5, 1);
BMduino_WS2812 BW(10, 8);
AHT10         myAHT10(AHT10_ADDRESS_0X38);

const int trigPin = 13;
const int echoPin = 12;
float    temp, humidity;
uint16_t duration, cm;

void dispTemp(int pos, float val) {
  int v = (int)(val * 10 + 0.5);
  lc.setChar(0, pos,   ' ', false);
  lc.setDigit(0, pos-1, (v/100)%10, false);
  lc.setDigit(0, pos-2, (v/10)%10,  true);
  lc.setDigit(0, pos-3, v%10,        false);
}

void dispInt(int pos, int val) {
  if (val > 9999) val = 9999;
  lc.setDigit(0, pos,   val/1000,       false);
  lc.setDigit(0, pos-1, (val/100)%10, false);
  lc.setDigit(0, pos-2, (val/10)%10,  false);
  lc.setDigit(0, pos-3, val%10,        false);
}

void setup() {
  Serial.begin(9600);
  Wire.begin();
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  lc.shutdown(0, false);
  lc.setIntensity(0, 9);
  lc.clearDisplay(0);
  BW.begin();

  while (myAHT10.begin() != true) {
    Serial.println("AHT10 not connected...");
    delay(5000);
  }
  Serial.println("Init OK");
}

void loop() {
  temp     = myAHT10.readTemperature();
  humidity = myAHT10.readHumidity();

  // 超音波測距
  digitalWrite(trigPin, LOW);
  delayMicroseconds(5);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  duration = pulseIn(echoPin, HIGH);
  cm       = duration / 58;

  Serial.print("Temp:");  Serial.print(temp);
  Serial.print("C  Dist:"); Serial.print(cm);
  Serial.println("cm");

  dispTemp(7, temp);
  dispInt(3,  cm);

  // WS2812：近=紅  中=綠  遠=藍
  if (cm < 20) {
    BW.fill(0, 10, 255, 0,   0);   // 紅
  } else if (cm < 100) {
    BW.fill(0, 10, 0,   255, 0);   // 綠
  } else {
    BW.fill(0, 10, 0,   0,   255); // 藍
  }
  BW.show();
  delay(500);
}

MAX7219 API

函數說明
lc.setDigit(dev, pos, digit, dp) 顯示數字 0–9。pos：7=最左 / 0=最右。dp=true 顯示小數點
lc.setChar(dev, pos, ch, dp) 顯示字元：' ' 空白、'-'、'H'、'E'、'L'、'0'–'9'
lc.clearDisplay(dev) 清除全部顯示
lc.setIntensity(dev, 0–15) 設定亮度，範例中使用 9
lc.shutdown(dev, false) 關閉省電模式（setup 必備）

WS2812 API

函數說明
BW.begin() 初始化（setup 呼叫）
BW.fill(start, count, R, G, B) 從第 start 顆填充 count 顆。R/G/B = 0–255
BW.show() 刷新顯示，改色後必須呼叫才有效

常用 WS2812 顏色

紅　255, 0, 0
綠　0, 255, 0
藍　0, 0, 255
黃　255, 255, 0
白　255, 255, 255
關　0, 0, 0
