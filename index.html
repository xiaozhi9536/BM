Arduino C++ · AHT10 + BH1750 
// ── AHT10（溫濕度）+ BH1750（光照）+ MAX7219 + WS2812 ──
// AHT10 VIN 接 3.3V！  WS2812 Data=D8（避免與 MAX CLK=D6 衝突）
// 須插麵包板：MAX7219、WS2812　不須：AHT10、BH1750（I2C）

#include <Wire.h>
#include <AHT10.h>
#include <BH1750.h>
#include <LedControl.h>
#include <BMduino_WS2812.h>

LedControl    lc = LedControl(7, 6, 5, 1); // Data=D7, CLK=D6, CS=D5
BMduino_WS2812 BW(10, 8);                  // 10顆, Data=D8
AHT10         myAHT10(AHT10_ADDRESS_0X38);
BH1750        lightMeter;

float    temp, humidity;
uint16_t lux;

void showStudentID() {
  // 顯示學號：4 b 3 3 0 0 8 3
  lc.setDigit(0, 7, 4,   false); // 4
  lc.setChar(0,  6, 'b', false); // b
  lc.setDigit(0, 5, 3,   false); // 3
  lc.setDigit(0, 4, 3,   false); // 3
  lc.setDigit(0, 3, 0,   false); // 0
  lc.setDigit(0, 2, 0,   false); // 0
  lc.setDigit(0, 1, 8,   false); // 8
  lc.setDigit(0, 0, 3,   false); // 3
  delay(3000);
  lc.clearDisplay(0);
  delay(500);
}

void dispTemp(int pos, float val) {
  int v = (int)(val * 10 + 0.5);
  lc.setChar(0,  pos,   ' ', false);
  lc.setDigit(0, pos-1, (v/100)%10, false);
  lc.setDigit(0, pos-2, (v/10)%10,  true); // 小數點
  lc.setDigit(0, pos-3, v%10,        false);
}

void dispInt(int pos, int val) {
  if (val > 9999) val = 9999;
  lc.setDigit(0, pos,   val/1000,       false);
  lc.setDigit(0, pos-1, (val/100)%10, false);
  lc.setDigit(0, pos-2, (val/10)%10,  false);
  lc.setDigit(0, pos-3, val%10,        false);
}

void setup() {
  Serial.begin(9600);
  Wire.begin();

  lc.shutdown(0, false);
  lc.setIntensity(0, 9);
  lc.clearDisplay(0);
  BW.begin();

  while (myAHT10.begin() != true) {
    Serial.println("AHT10 not connected...");
    delay(5000);
  }
  lightMeter.begin();

  showStudentID(); // ← 開機顯示學號 4b330083
}

void loop() {
  temp     = myAHT10.readTemperature();
  humidity = myAHT10.readHumidity();
  lux      = lightMeter.readLightLevel();

  Serial.print("Temp:");  Serial.print(temp);
  Serial.print("C  Hum:"); Serial.print(humidity);
  Serial.print("%  Lux:"); Serial.println(lux);

  dispTemp(7, temp); // 左4位
  dispInt(3,  lux);  // 右4位

  if      (lux < 100) { BW.fill(0,10,0,0,255); } // 藍
  else if (lux < 500) { BW.fill(0,10,0,255,0); } // 綠
  else               { BW.fill(0,10,255,0,0); } // 紅
  BW.show();
  delay(1000);
}

MAX7219 API

函數說明
lc.setDigit(dev,pos,digit,dp)顯示數字 0–9。pos：7=最左 / 0=最右。dp=true 顯示小數點
lc.setChar(dev,pos,ch,dp)字元：' ' '-' 'b' 'H' 'E' 'L' 'P' 等
lc.clearDisplay(dev)清除全部
lc.setIntensity(dev,0–15)亮度，範例使用 9
lc.shutdown(dev,false)關閉省電模式（setup 必備）
